function parseTelegramToCSV(inputText) { const lines = inputText.split("\n").map(l => l.trim()).filter(l => l.length > 0); let category = ""; let id = 1; const rows = []; const categoryEmojiRegex = /^[📱🖥💻⌚🎧]\s*(.+)$/i; const categoryPlainRegex = /^(?!.*\d+\s*BYN)(?!.*\b(GB|TB)\b)(?!.*\b(esim|sim|wifi|cellular)\b).+$/i; const productRegex = /^(.+?)\s+(\d+(?:TB|GB)?)\s+([\wА-Яа-я+]+)\s+([\w+]+)\s*-\s*(\d+)\s*BYN/i; function detectModel(category, namePart) { const base = category.split("/")[0].trim(); // например: "iPhone 17 Pro" const words = base.split(" "); // Определяем тип устройства const deviceType = words[0]; // iPhone / iPad / MacBook / Watch / AirPods // Определяем номер модели из категории const modelNumber = words[1] || ""; // Приводим namePart к нормальному виду const cleaned = namePart .replace(/pro/i, "Pro") .replace(/max/i, "Max") .replace(/air/i, "Air") .replace(/plus/i, "Plus") .replace(/ultra/i, "Ultra") .trim(); return `${deviceType} ${cleaned}`; } for (const line of lines) { const emojiMatch = line.match(categoryEmojiRegex); if (emojiMatch) { category = emojiMatch[1].trim(); continue; } const plainMatch = line.match(categoryPlainRegex); if (plainMatch) { category = plainMatch[0].trim(); continue; } const prodMatch = line.match(productRegex); if (prodMatch) { const name = prodMatch[1].trim(); const memory = prodMatch[2].trim(); const color = prodMatch[3].trim(); const sim = prodMatch[4].trim(); const price = prodMatch[5].trim(); const model = detectModel(category, name); rows.push({ id: id++, category, model, name, memory, color, sim, price }); } } let csv = "id,category,model,name,memory,color,sim,price\n"; rows.forEach(r => { csv += `${r.id},${escapeCSV(r.category)},${escapeCSV(r.model)},${escapeCSV(r.name)},${escapeCSV(r.memory)},${escapeCSV(r.color)},${escapeCSV(r.sim)},${r.price}\n`; }); return csv; }
